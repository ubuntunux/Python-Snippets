## 폴더 용량 알아내기
    import os
    from os.path import join, getsize
    for root, dirs, files in os.walk('.'):
        result = "%s : %.f MB in %d files." % (os.path.abspath(root), sum([getsize(join(root, name)) for name in files]) / (1024.0 * 1024.0), len(files))
        print result
